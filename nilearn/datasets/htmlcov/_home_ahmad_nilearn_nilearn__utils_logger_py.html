11# The technique used in the log() function only applies to CPython, because 

12# it uses the inspect module to walk the call stack. 

13def log(msg, verbose=1, object_classes=(BaseEstimator, ), 

14 stack_level=1, msg_level=1): 

15 """Display a message to the user, depending on the verbosity level. 

16 

17 This function allows to display some information that references an object 

18 that is significant to the user, instead of a internal function. The goal 

19 is to make user's code as simple to debug as possible. 

20 

21 Parameters 

22 ---------- 

23 msg : str 

24 Message to display. 

25 

26 verbose : int, optional 

27 Current verbosity level. Message is displayed if this value is greater 

28 or equal to msg_level. Default=1. 

29 

30 object_classes : tuple of type, optional 

31 Classes that should appear to emit the message. 

32 Default=(BaseEstimator, ). 

33 

34 stack_level : int, optional 

35 If no object in the call stack matches object_classes, go back that 

36 amount in the call stack and display class/function name thereof. 

37 Default=1. 

38 

39 msg_level : int, optional 

40 Verbosity level at and above which message should be displayed to the 

41 user. Most of the time this parameter can be left unchanged. 

42 Default=1. 

43 

44 Notes 

45 ----- 

46 This function does tricky things to ensure that the proper object is 

47 referenced in the message. If it is called e.g. inside a function that is 

48 called by a method of an object inheriting from any class in 

49 object_classes, then the name of the object (and the method) will be 

50 displayed to the user. If several matching objects exist in the call 

51 stack, the highest one is used (first call chronologically), because this 

52 is the one which is most likely to have been written in the user's script. 

53 

54 """ 

55 if verbose >= msg_level: 

56 stack = inspect.stack() 

57 object_frame = None 

58 object_self = None 

59 for f in reversed(stack): 

60 frame = f[0] 

61 current_self = frame.f_locals.get("self", None) 

62 if isinstance(current_self, object_classes): 

63 object_frame = frame 

64 func_name = f[3] 

65 object_self = current_self 

66 break 

67 

68 if object_frame is None: # no object found: use stack_level 

69 if stack_level >= len(stack): 

70 stack_level = -1 

71 func_name = '<top_level>' 

72 else: 

73 object_frame, _, _, func_name = stack[stack_level][:4] 

74 object_self = object_frame.f_locals.get("self", None) 

75 

76 if object_self is not None: 

77 func_name = "%s.%s" % (object_self.__class__.__name__, func_name) 

78 

79 

80 print("[{func_name}] {msg}".format(func_name=func_name, msg=msg))

83def _compose_err_msg(msg, **kwargs): 

84 """Append key-value pairs to msg, for display. 

85 

86 Parameters 

87 ---------- 

88 msg : string 

89 Arbitrary message. 

90 

91 kwargs : dict, optional 

92 Arbitrary dictionary. 

93 

94 Returns 

95 ------- 

96 updated_msg : string 

97 msg, with "key: value" appended. Only string values are appended. 

98 

99 Example 

100 ------- 

101 >>> _compose_err_msg('Error message with arguments...', arg_num=123, \ 

102 arg_str='filename.nii', arg_bool=True) 

103 'Error message with arguments...\\narg_str: filename.nii' 

104 >>> 

105 

106 """ 

107 updated_msg = msg 

108 for k, v in sorted(kwargs.items()): 

109 if isinstance(v, str): # print only str-like arguments 

110 updated_msg += "\n" + k + ": " + v 

111 

112 return updated_msg
