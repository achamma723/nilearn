1279def _load_mixed_gambles(zmap_imgs): 

1280 """Ravel zmaps (one per subject) along time axis, resulting, 

1281 in a n_subjects * n_trials 3D niimgs and, and then make 

1282 gain vector y of same length. 

1283 """ 

1284 X = [] 

1285 y = [] 

1286 mask = [] 

1287 for zmap_img in zmap_imgs: 

1288 # load subject data 

1289 this_X = get_data(zmap_img) 

1290 affine = zmap_img.affine 

1291 finite_mask = np.all(np.isfinite(this_X), axis=-1) 

1292 this_mask = np.logical_and(np.all(this_X != 0, axis=-1), 

1293 finite_mask) 

1294 this_y = np.array([np.arange(1, 9)] * 6).ravel() 

1295 

1296 # gain levels 

1297 if len(this_y) != this_X.shape[-1]: 

1298 raise RuntimeError("%s: Expecting %i volumes, got %i!" % ( 

1299 zmap_img, len(this_y), this_X.shape[-1])) 

1300 

1301 # standardize subject data 

1302 this_X -= this_X.mean(axis=-1)[..., np.newaxis] 

1303 std = this_X.std(axis=-1) 

1304 std[std == 0] = 1 

1305 this_X /= std[..., np.newaxis] 

1306 

1307 # commit subject data 

1308 X.append(this_X) 

1309 y.extend(this_y) 

1310 mask.append(this_mask) 

1311 y = np.array(y) 

1312 X = np.concatenate(X, axis=-1) 

1313 mask = np.sum(mask, axis=0) > .5 * len(mask) 

1314 mask = np.logical_and(mask, np.all(np.isfinite(X), axis=-1)) 

1315 X = X[mask, :].T 

1316 tmp = np.zeros(list(mask.shape) + [len(X)]) 

1317 tmp[mask, :] = X.T 

1318 mask_img = nibabel.Nifti1Image(mask.astype(int), affine) 

1319 X = nibabel.four_to_three(nibabel.Nifti1Image(tmp, affine)) 

1320 return X, y, mask_img

2316def select_from_index(urls, inclusion_filters=None, exclusion_filters=None, 

2317 n_subjects=None): 

2318 """Select subset of urls with given filters. 

2319 

2320 Parameters 

2321 ---------- 

2322 urls : list of str 

2323 List of dataset urls obtained from index download. 

2324 

2325 inclusion_filters : list of str, optional 

2326 List of unix shell-style wildcard strings 

2327 that will be used to filter the url list. 

2328 If a filter matches the url it is retained for download. 

2329 Multiple filters work on top of each other. 

2330 Like an "and" logical operator, creating a more restrictive query. 

2331 Inclusion and exclusion filters apply together. 

2332 For example the filter '*task-rest*'' would keep only urls 

2333 that contain the 'task-rest' string. 

2334 

2335 exclusion_filters : list of str, optional 

2336 List of unix shell-style wildcard strings 

2337 that will be used to filter the url list. 

2338 If a filter matches the url it is discarded for download. 

2339 Multiple filters work on top of each other. 

2340 Like an "and" logical operator, creating a more restrictive query. 

2341 Inclusion and exclusion filters apply together. 

2342 For example the filter '*task-rest*' would discard all urls 

2343 that contain the 'task-rest' string. 

2344 

2345 n_subjects : int, optional 

2346 Number of subjects to download from the dataset. All by default. 

2347 

2348 Returns 

2349 ------- 

2350 urls : list of string 

2351 Sorted list of filtered dataset directories. 

2352 

2353 """ 

2354 inclusion_filters = inclusion_filters if inclusion_filters else [] 

2355 exclusion_filters = exclusion_filters if exclusion_filters else [] 

2356 # We apply filters to the urls 

2357 for exclusion in exclusion_filters: 

2358 urls = [url for url in urls if not fnmatch.fnmatch(url, exclusion)] 

2359 for inclusion in inclusion_filters: 

2360 urls = [url for url in urls if fnmatch.fnmatch(url, inclusion)] 

2361 

2362 # subject selection filter 

2363 # from the url list we infer all available subjects like 'sub-xxx/' 

2364 subject_regex = 'sub-[a-z|A-Z|0-9]*[_./]' 

2365 

2366 def infer_subjects(urls): 

2367 subjects = set() 

2368 for url in urls: 

2369 if 'sub-' in url: 

2370 subjects.add(re.search(subject_regex, url).group(0)[:-1]) 

2371 return sorted(subjects) 

2372 

2373 # We get a list of subjects (for the moment the first n subjects) 

2374 selected_subjects = set(infer_subjects(urls)[:n_subjects]) 

2375 # We exclude urls of subjects not selected 

2376 urls = [ 

2377 url for url in urls 

2378 if 'sub-' not in url or re.search(subject_regex, url).group(0)[:-1] 

2379 in selected_subjects 

2380 ]
